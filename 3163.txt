var compressedString = function(word) {
    let comp = '';
    count = 1;
    const newWord = word.split('')
    if(newWord.length > 0){
        for (let i = 0; i < newWord.length; i++) {
            if(newWord[i] === newWord[i+1] && count < 9){
                count++;
            } else {
                comp += count + newWord[i];
                count = 1;
            }
        }
    }
    return comp;
};
